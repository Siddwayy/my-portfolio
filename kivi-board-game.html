Kivi Board Game (Java)

KiviBoardGame delivers a full Java Swing adaptation of the dice strategy board game Kivi: 2–4 players (human/AI) roll to match patterns on a 7×7 grid, build scoring lines over 10 rounds, with save/load and color vision support.

Dice Pattern Validation

/**
 * Validates dice combination against game rules.
 * Centralized validation shields rules from ripple changes (GRASP: Protected Variations).
 */
public boolean validateDiceCombination(int[] dice) {
    Arrays.sort(dice);
    int sum = Arrays.stream(dice).sum();
    
    // Sum over 30 pattern
    if (sum > 30) return true;
    
    // Three pairs pattern
    Map<Integer, Integer> counts = new HashMap<>();
    for (int die : dice) {
        counts.put(die, counts.getOrDefault(die, 0) + 1);
    }
    long pairCount = counts.values().stream()
        .filter(count -> count >= 2).count();
    if (pairCount == 3) return true;
    
    // Straight pattern (1-6)
    boolean isStraight = true;
    for (int i = 0; i < dice.length; i++) {
        if (dice[i] != i + 1) {
            isStraight = false;
            break;
        }
    }
    return isStraight;
}

Game State Serialization

/**
 * Saves complete game state using Java serialization.
 * Handles transient fields (threads) automatically.
 */
public void saveGame(String filename) throws IOException {
    try (ObjectOutputStream oos = new ObjectOutputStream(
            new FileOutputStream(filename))) {
        oos.writeObject(gameManager);  // Serializes board, players, round
        oos.writeObject(currentPlayerIndex);
        oos.writeObject(gameState);
    }
}

/**
 * Loads game state and restores UI.
 */
public void loadGame(String filename) throws IOException, 
        ClassNotFoundException {
    try (ObjectInputStream ois = new ObjectInputStream(
            new FileInputStream(filename))) {
        gameManager = (GameManager) ois.readObject();
        currentPlayerIndex = (Integer) ois.readObject();
        gameState = (GameState) ois.readObject();
        restoreUI();  // Rebuild UI from loaded state
    }
}

Threaded AI Player Logic

/**
 * AI runs on separate thread to prevent UI freeze (GRASP: Pure Fabrication).
 * Easy mode: 2 rolls max, random placement.
 * Hard mode: 3 rolls, blocks opponent threats.
 */
public class ComputerPlayerAI extends Thread {
    private Difficulty difficulty;
    private GameManager gameManager;
    
    @Override
    public void run() {
        int maxRolls = (difficulty == Difficulty.EASY) ? 2 : 3;
        int rolls = 0;
        
        while (rolls < maxRolls) {
            int[] dice = rollDice();
            List<Cell> validCells = findValidCells(dice);
            
            if (difficulty == Difficulty.HARD) {
                // Block opponent's longest line
                validCells = prioritizeBlocking(validCells);
            }
            
            if (!validCells.isEmpty()) {
                placeStone(validCells.get(0));
                break;
            }
            rolls++;
        }
        
        SwingUtilities.invokeLater(() -> {
            gameManager.endTurn();
        });
    }
}

Real-time Grid Pattern Matching

/**
 * Highlights valid cells when dice are rolled.
 * Decoupled via GridCellClickListener (Low Coupling).
 */
public void highlightValidCells(int[] dice) {
    clearHighlights();
    
    for (int row = 0; row < GRID_SIZE; row++) {
        for (int col = 0; col < GRID_SIZE; col++) {
            Cell cell = grid[row][col];
            if (cell.isEmpty() && matchesPattern(cell, dice)) {
                cell.setHighlighted(true);
                cell.setBorder(BorderFactory.createLineBorder(
                    Color.GREEN, 3));
            }
        }
    }
    repaint();
}

private boolean matchesPattern(Cell cell, int[] dice) {
    // Check if dice combination matches cell's required pattern
    return gameManager.validateDiceCombination(dice) &&
           cell.getPattern().matches(dice);
}

GameManager Controller Pattern

/**
 * Central controller managing game flow (GRASP: Controller).
 * Coordinates turns, scoring, and state transitions.
 */
public class GameManager implements Serializable {
    private Board board;
    private List<Player> players;
    private int currentRound;
    private int currentPlayerIndex;
    
    public void processTurn(int[] dice, int row, int col) {
        if (!validateDiceCombination(dice)) return;
        if (!board.isValidPlacement(row, col, dice)) return;
        
        Player currentPlayer = players.get(currentPlayerIndex);
        board.placeStone(row, col, currentPlayer.getColor());
        updateScores();
        
        if (currentPlayerIndex < players.size() - 1) {
            currentPlayerIndex++;
        } else {
            currentPlayerIndex = 0;
            currentRound++;
        }
        
        notifyListeners();  // Update UI
    }
}
